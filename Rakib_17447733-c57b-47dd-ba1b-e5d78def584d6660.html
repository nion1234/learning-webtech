https:coffee.js

function getCost(){
    var frenchQty = document.getElementById('french').value; /* get values from html form by ID */
    var hazlenutQty = document.getElementById('hazlenut').value;
    var colombianQty = document.getElementById('colombian').value;

    var frenchPrice = 3.49; /* price of product for calculations */
    var hazlenutPrice = 3.95;
    var colombianPrice = 4.49;

    var french = frenchQty * frenchPrice; /* multiplication of quantity with price */
    var hazlenut = hazlenutQty * hazlenutPrice;
    var colombian = colombianQty * colombianPrice;
    
    var totalCost = french+hazlenut+colombian; /* total cost addition */
    
    document.getElementById('cost').value="$"+totalCost; /* displaying total cose to input box whos id is cost */
}
